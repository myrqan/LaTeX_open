\tcbset{
  dashedbox/.style={
    breakable,
    enhanced,
    frame hidden,
    borderline={0.3pt}{0pt}{blue!50!white,dashed},
    %fontupper = \fontsize{9pt}{8pt}\selectfont,
    %before upper = {\everymath{\scriptstyle}},
    colback = white,
    arc = 0mm,
    outer arc = 0mm,
    boxsep = 0.5pt,
    leftrule=-6pt,
    rightrule=-6pt
  }
}
